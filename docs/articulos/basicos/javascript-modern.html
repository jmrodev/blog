JavaScript Moderno: ES6+ y más allá

Introducción

JavaScript ha evolucionado significativamente desde su creación. Las versiones modernas (ES6+) han introducido características que han transformado la forma en que escribimos código.

Características principales de ES6+

1. Arrow Functions

Sintaxis más concisa para funciones:

// Función tradicional
function sumar(a, b) {
    return a + b;
}

// Arrow function
const sumar = (a, b) => a + b;

// Arrow function con múltiples líneas
const procesarDatos = (datos) => {
    const resultado = datos.map(item => item * 2);
    return resultado.filter(item => item > 10);
};

2. Destructuring

Extraer valores de objetos y arrays:

// Destructuring de objetos
const usuario = { nombre: 'Juan', edad: 30, ciudad: 'Madrid' };
const { nombre, edad } = usuario;
console.log(nombre); // 'Juan'

// Destructuring de arrays
const colores = ['rojo', 'verde', 'azul'];
const [primero, segundo] = colores;
console.log(primero); // 'rojo'

// Destructuring en parámetros de función
const mostrarUsuario = ({ nombre, edad }) => {
    console.log(`${nombre} tiene ${edad} años`);
};

3. Template Literals

Strings con interpolación:

const nombre = 'María';
const edad = 25;

// Antes
const mensaje = nombre + ' tiene ' + edad + ' años';

// Ahora
const mensaje = `${nombre} tiene ${edad} años`;

// Multilínea
const html = `
    <div class="usuario">
        <h2>${nombre}</h2>
        <p>Edad: ${edad}</p>
    </div>
`;

4. Spread y Rest Operators

// Spread operator
const numeros = [1, 2, 3];
const masNumeros = [...numeros, 4, 5]; // [1, 2, 3, 4, 5]

const persona = { nombre: 'Ana', edad: 28 };
const personaCompleta = { ...persona, ciudad: 'Barcelona' };

// Rest operator
const sumar = (...args) => {
    return args.reduce((total, num) => total + num, 0);
};

console.log(sumar(1, 2, 3, 4, 5)); // 15

5. Promesas y Async/Await

// Promesas
const obtenerDatos = () => {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            const datos = { id: 1, nombre: 'Producto' };
            resolve(datos);
        }, 1000);
    });
};

// Async/Await
const cargarDatos = async () => {
    try {
        const datos = await obtenerDatos();
        console.log('Datos cargados:', datos);
        return datos;
    } catch (error) {
        console.error('Error:', error);
    }
};

// Múltiples promesas
const cargarTodo = async () => {
    const [usuarios, productos] = await Promise.all([
        fetch('/api/usuarios'),
        fetch('/api/productos')
    ]);
    
    return {
        usuarios: await usuarios.json(),
        productos: await productos.json()
    };
};

6. Módulos ES6

// math.js
export const sumar = (a, b) => a + b;
export const restar = (a, b) => a - b;

export default class Calculadora {
    multiplicar(a, b) {
        return a * b;
    }
}

// main.js
import Calculadora, { sumar, restar } from './math.js';

const calc = new Calculadora();
console.log(sumar(5, 3)); // 8
console.log(calc.multiplicar(4, 2)); // 8

7. Clases

class Usuario {
    constructor(nombre, email) {
        this.nombre = nombre;
        this.email = email;
    }
    
    saludar() {
        return `¡Hola, soy ${this.nombre}!`;
    }
    
    static crearAdmin(nombre, email) {
        const admin = new Usuario(nombre, email);
        admin.esAdmin = true;
        return admin;
    }
}

class Empleado extends Usuario {
    constructor(nombre, email, salario) {
        super(nombre, email);
        this.salario = salario;
    }
    
    obtenerInfo() {
        return `${this.saludar()} Mi salario es ${this.salario}`;
    }
}

Características más recientes

Optional Chaining (ES2020)

const usuario = {
    perfil: {
        direccion: {
            calle: 'Calle Principal'
        }
    }
};

// Antes
const calle = usuario.perfil && usuario.perfil.direccion && usuario.perfil.direccion.calle;

// Ahora
const calle = usuario.perfil?.direccion?.calle;

Nullish Coalescing (ES2020)

const valor = null;
const valorPorDefecto = valor ?? 'valor por defecto';
console.log(valorPorDefecto); // 'valor por defecto'
